\newcommand\dedication[1]{%
	\thispagestyle{empty}%

	\begin{flushright}%
		#1%

\newcommand\epigraph[3][2\baselineskip]{%
	\begin{flushright}%
		{\rmfamily\itshape #2}%
		\vskip 0.5\baselineskip%
		#3%
	\end{flushright}%
	\vspace{#1}%
}
